```html
<!-- Button.vue -->

<template>
  <button type="button" :disabled="isDisabled">{{ label }}</button>
</template>

<script>
  export default {
    name: 'button',
    props: {
      /**
       * Checks if the button should be disabled
       */
      isDisabled: {
        type: Boolean,
        default: false,
        required: true,
      },
      /**
       * The display label of the button
       */
      label: {
        type: String,
        default: 'One',
        required: true,
      },
    },
  };
</script>
```
